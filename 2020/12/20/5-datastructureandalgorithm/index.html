        area = (j-i)*Math.min(height[i], height[j]);

  4. 数组(三数之和)—高频老题

    题述：给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有满足条件且不重复的三元组。

    注意：答案中不可以包含重复的三元组。

    示例：

    给定数组 nums = [-1, 0, 1, 2, -1, -4]，

满足要求的三元组集合为：
[
  [-1, 0, 1],
  [-1, -1, 2]
]

    实现：

    // 暴力解法，依次枚举三个数求和，和为0则添加至结果集，有重复则跳出本次循环
public List<List<Integer>> threeSum(int[] nums) {
    Arrays.sort(nums);
    List<List<Integer>> sum = new LinkedList<>();
    // 遍历寻找第1个数
    for (int i = 0; i < nums.length; i++) {
        // 遍历寻找第2个数
        for (int j = i + 1; j < nums.length; j++) {
            // 遍历寻找第3个数
            for (int k = j + 1; k < nums.length; k++) {
                // 如果三个数的和相加等于0；则将三数添加至sum数组
                if ((nums[i] + nums[j] + nums[k]) == 0) {
                    // 如果结果集已有该和为零的序列，则结束本次循环；继续下一次
                    if (sum.contains(Arrays.asList(nums[i], nums[j], nums[k]))) {
                        continue;
                    }
                    // 为结果集合添加符合要求的数组
                    sum.add(Arrays.asList(nums[i], nums[j], nums[k]));
                }
            }
        }
    }
    // 返回最终的结果集
    return sum;
}

// 对原始数组进行排序
Arrays.sort(nums);
// 初始化结果集

// 枚举所有的变量

        int hi = nums.length - 1;

        int sum = 0 - nums[i];
        // 如果左端序列比右端序列小则一直进行查找
        while (lo < hi) {
            // 如果在左右端找出和为sum的值即将目标值，左端值，右端值添加到结果集合
            if (nums[lo] + nums[hi] == sum) {
                // 将符合条件的目标值，左端值和右端值添加至结果集
                res.add(Arrays.asList(nums[i], nums[lo], nums[hi]));
                // 如果左端序列小于右端序列；且；左端值等于左端序列之前的值就跳过该序列
                while (lo < hi && nums[lo] == nums[lo + 1]) {
                    lo++;
                }
                // 如果左端序列小于右端序列；且；右端值等于右端序列之后的值就跳过该序列
                while (lo < hi && nums[hi] == nums[hi - 1]) {
                    hi--;
                }
                // 左端序列递增
                lo++;
                // 右端序列递减
                hi--;
            } else if (nums[lo] + nums[hi] < sum) {// 如果三数和小于0，则继续向前找左端值
                lo++;
            } else {// 如果三数和大于0，则继续向后找右端值
                hi--;
            }
        }
    }
}

    题述：反转一个单链表

    示例：

    输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL

    // 迭代解法
public ListNode reverseList(ListNode head) {
    // 初始化前驱节点
    ListNode prevHead = null;
    // 如果头结点不为null
    while (head != null) {
        // 记录头结点的指针
        ListNode recordNext = head.next;
        // 将头结点的指针指向前驱节点
        head.next = prevHead;
        // 将头结点和前驱节点值交换
        prevHead = head;
        head = recordNext;
    }
    // 返回最新链表
    return prevHead;
}
    // 递归解法一
public ListNode reverseList(ListNode head) {
    return reverseListInt(head, null);
}

private ListNode reverseListInt(ListNode head, ListNode newHead) {
    if (head == null) {
        // 如果头结点的指针指向null说明链表翻转完成，则返回反转后的链表
        return newHead;
    }
    // 交换头结点和前驱节点的值
    ListNode next = head.next;
    head.next = newHead;
    // 将交换后的指针和头结点再次进行交换；直到头结点的指针指向null后返回最新链表
    return reverseListInt(next, head);
}
    // 递归解法二
public ListNode reverseList(ListNode head) {
    if (head == null || head.next == null) {
        // 如果头结点的指针指向null说明链表翻转完成，则返回反转后的链表
        return head;
    }
    // 交换头结点和递归交换前驱节点的值
    ListNode nextNode = head.next;
    ListNode newHead = reverseList(nextNode);
    // 将前驱节点的指针指向头结点
    nextNode.next = head;
    // 将头结点的指针指向null
    head.next = null;
    // 返回最新的链表
    return newHead;
}

  7. 链表(两两交换链表中的节点)

    题述：给定一个链表，两两交换其中相邻的节点，并返回交换后的链表。你不能只是单纯的改变节点内部的值，而是需要实际的进行节点交换。

    示例：

    输入：head = [1,2,3,4]
输出：[2,1,4,3]

输入：head = []
输出：[]

输入：head = [1]
输出：[1]

    提示：

    • 链表中节点的数目在范围 [0, 100] 内
    • 0 <= Node.val <= 100

    实现：

    // 递归解法
public ListNode swapPairs(ListNode head) {
    if ((head == null) || (head.next == null)) {
        return head;
    }
    // 记录头结点指针的位置
    ListNode prevHead = head.next;
    // 递归将头节点的指针指向头结点指针的指针
    head.next = swapPairs(head.next.next);
    // 将前驱元素的指针指向头结点
    prevHead.next = head;
    // 返回最新链表
    return prevHead;
}
    // 迭代解法
private static void algorithm(ListNode head) {
    ListNode dummy = new ListNode(0);
    // 将链表的指针指向头结点
    dummy.next = head;
    // 记录当前链表
    ListNode current = dummy;
    // 如果当前链表的指针不为null且者前驱元素的指针不为null
    while (current.next != null && current.next.next != null) {
        // 记录头结点的指针
        ListNode first = current.next;
        // 记录头结点前驱元素的指针
        ListNode second = current.next.next;
        // 将头结点前驱元素的指针指向头结点前驱的前驱的指针
        first.next = second.next;
        // 将头结点的指针指向头结点前驱元素的指针
        current.next = second;
        // 将头结点前去元素的指针指向头结点的指针
        current.next.next = first;
        // 将当前链表指向头结点前去元素的指针
        current = current.next.next;
    }
    // 返回头结点前驱元素的指针
    return dummy.next;
}

  8. 链表(环形链表)

    题述：给定一个链表，判断链表中是否有环。

    如果链表中有某个节点，可以通过连续跟踪 next 指针再次到达，则链表中存在环。 为了表示给定链表中的环，我们使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。 如果 pos 是 -1，则在该链表中没有环。注意：pos 不作为参数进行传递，仅仅是为了标识链表的实际情况。

    如果链表中存在环，则返回 true 。 否则，返回 false 。

    附加：

    你能用 *O(1)*（即，常量）内存解决此问题吗？

    示例：

    输入：head = [3,2,0,-4], pos = 1
输出：true
解释：链表中有一个环，其尾部连接到第二个节点。

    输入：head = [1,2], pos = 0
输出：true
解释：链表中有一个环，其尾部连接到第一个节点。

    输入：head = [1], pos = -1
输出：false
解释：链表中没有环。

    提示：

    • 链表中节点的数目范围是 [0, 104]
    • -105 <= Node.val <= 105
    • pos 为 -1 或者链表中的一个 有效索引 。

    题解：

    1. 使用两个指针，一个快指针，一个慢指针
    2. 慢指针一次走一步，快指针一次走两步
    3. 当快慢指针指向同一个元素说明有环存在，否则无环存在

    实现：

    // 快慢指针解法
public boolean hasCycle(ListNode head) {
    if (head == null) {
        return false;
    }
    // 初始化慢指针
    ListNode walker = head;
    // 初始化快指针
    ListNode runner = head;
    // 如果快指针不指向null且快指针的前驱指针不为null
    while (runner.next != null && runner.next.next != null) {
        // 将慢指针指向他的前驱指针
        walker = walker.next;
        // 将快指针指向他的前驱指针
        runner = runner.next.next;
        // 如果快慢指针指向同一个元素说明有环存在
        if (walker == runner) {
            return true;
        }
    }
    // 快慢指针指向不是同一个元素
    return false;  
}

  9. 链表(环形链表 II)

    题述：给定一个链表，返回链表开始入环的第一个节点。 如果链表无环，则返回 null。

    为了表示给定链表中的环，我们使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。 如果 pos 是 -1，则在该链表中没有环。注意，pos 仅仅是用于标识环的情况，并不会作为参数传递到函数中。

    说明：不允许修改给定的链表。

    进阶：

    你是否可以使用 O(1) 空间解决此题？

    示例：

    输入：head = [3,2,0,-4], pos = 1
输出：返回索引为 1 的链表节点
解释：链表中有一个环，其尾部连接到第二个节点。

    输入：head = [1,2], pos = 0
输出：返回索引为 0 的链表节点
解释：链表中有一个环，其尾部连接到第一个节点。

    输入：head = [1], pos = -1
输出：返回 null
解释：链表中没有环。

    提示：

    • 链表中节点的数目范围在范围 [0, 104] 内
    • -105 <= Node.val <= 105
    • pos 的值为 -1 或者链表中的一个有效索引

    实现：

    public ListNode detectCycle(ListNode head) {
    if (head == null || head.next == null) {
        return null;
    }
    // 初始化慢指针
    ListNode slow = head;
    // 初始化快指针
    ListNode fast = head;
    // 如果快指针不指向null且快指针的前驱指针不为null
    while (fast != null && fast.next != null) {
        // 将慢指针指向其前驱的指针
        slow = slow.next;
        // 将快指针指向前驱的前驱
        fast = fast.next.next;
        // 如果快慢指针相遇，则说明链表有环
        if (slow == fast) {
            // 如果慢指针没有指向头结点
            while (slow != head) {
                // 就一直往前走
                slow = slow.next;
                // 将头结点指向自身
                head = head.next;
            }
            // 返回头结点
            return head;
        }
    }
    return null;
}

  10. 链表(K 个一组翻转链表)

    题述：给你一个链表，每 k 个节点一组进行翻转，请你返回翻转后的链表。

    k 是一个正整数，它的值小于或等于链表的长度。

    如果节点总数不是 k 的整数倍，那么请将最后剩余的节点保持原有顺序。

    示例：

    输入: head = [1,2,3,4,5], k = 2
输出: [2,1,4,3,5]

    输入: head = [1,2,3,4,5], k = 3
输出: [3,2,1,4,5]

输入: head = [1,2,3,4,5], k = 1
输出: [1,2,3,4,5]

输入: head = [1], k = 1
输出: [1]

    说明：

    • 你的算法只能使用常数的额外空间。
    • 你不能只是单纯的改变节点内部的值，而是需要实际进行节点交换。

    提示：

    1. 列表中节点的范围是sz.
    2. 1 <= sz <= 5000
    3. 0 <= Node.val <= 1000
    4. 1 <= k <= sz

    实现：

    // 递归解法
public ListNode reverseKGroup(ListNode head, int k) {
    ListNode curr = head;
    int count = 0;
    while (curr != null && count != k) { // find the k+1 node
        curr = curr.next;
        count++;
    }
    if (count == k) { // if k+1 node is found
        curr = reverseKGroup(curr, k); // reverse list with k+1 node as head
        // head - head-pointer to direct part, 
        // curr - head-pointer to reversed part;
        while (count-- > 0) { // reverse current k-group:
            ListNode tmp = head.next; // tmp - next head in direct part
            head.next = curr; // preappending "direct" head to the reversed list
            curr = head; // move head of reversed part to a new node
            head = tmp; // move "direct" head to the next node in direct part
        }
        head = curr;
    }
    return head;
}
    // 迭代解法
public ListNode reverseKGroup(ListNode head, int k) {
    int n = 0;
    for (ListNode i = head; i != null; n++, i = i.next);

    ListNode dmy = new ListNode(0);
    dmy.next = head;
    for(ListNode prev = dmy, tail = head; n >= k; n -= k) {
        for (int i = 1; i < k; i++) {
            ListNode next = tail.next.next;
            tail.next.next = prev.next;
            prev.next = tail.next;
            tail.next = next;
        }

        prev = tail;
        tail = tail.next;
    }
    return dmy.next;
}

            head = begin.next;
        } else {

            head = head.next;
        }
    }

    return dummyhead.next;

}

public ListNode reverse(ListNode begin, ListNode end){
    // 将开始元素的指针记录
    ListNode curr = begin.next;
    // 定义两个链表
    ListNode next, first;
    // 将开始元素赋记录
    ListNode prev = begin;
    // 将开始元素的指针指向反转后的链表
    first = curr;
    // 如果反转链表的值和原始链表的结束值不一致，就一直进行交换操作
    while (curr!=end){
        // 将反转链表的指针指向原始链表
        next = curr.next;
        // 将反转链表的指针指向开始元素的值
        curr.next = prev;
        // 用反转链表替换原始元素
        prev = curr;
        // 用原始链表替换反转链表
        curr = next;
    }
    // 将开始元素的指针指向最终的元素
    begin.next = prev;
    // 将反转链表的指针指向开始元素的值
    first.next = curr;
    // 返回最终的反转链表
    return first;
}

    题述：给定一个排序数组，你需要在 原地 删除重复出现的元素，使得每个元素只出现一次，返回移除后数组的新长度。

    示例：

    给定数组 nums = [1,1,2],

你不需要考虑数组中超出新长度后面的元素。

函数应该返回新的长度 5, 并且原数组 nums 的前五个元素被修改为 0, 1, 2, 3, 4。
你不需要考虑数组中超出新长度后面的元素。

    说明:

    为什么返回数值是整数，但输出的答案是数组呢?

    请注意，输入数组是以「引用」方式传递的，这意味着在函数里修改输入数组对于调用者是可见的。

// 根据你的函数返回的长度, 它会打印出数组中该长度范围内的所有元素。

}

    // 简单解法
public int removeDuplicates(int[] nums) {
    // 初始化重复数字的计数器
    int count = 0;
    // 获取数组的长度
    int len = nums.length;
    // 遍历数组拿出每一个元素
    for (int i = 1; i < len; i++) {
        // 如果当前元素和上一个元素一致，则计数器值+1
        if (nums[i] == nums[i - 1]) {
            count++;
        } else {
            // 否则将当前元素赋给去重后的最新索引位置的值
            nums[i - count] = nums[i];
        }
    }
    // 返回最新的数组长度
    return len - count;
}
    // Time O(n), Space O(1)

    // 获取数组长度
    int len = nums.length;
    // 如果长度小于2直接返回当前数组的长度
    if (len < 2) {
        return len;

    // 初始化非重数组长度计数器
    int id = 1;

        if (nums[i] != nums[i - 1]) {
            nums[id++] = nums[i];
        }
    }
    // 返回非重复元素数组的长度
    return id;
}

            nums[length++] = num;
        }
    // 返回非重数组的长度
    return length;
}

    说明:

    • 尽可能想出更多的解决方案，至少有三种不同的方法可以解决这个问题。

    public void rotate(int[] nums, int k) {
    // 获取k的索引值
    k %= nums.length;
    // 执行旋转
    reverse(nums, 0, nums.length - 1);
    reverse(nums, 0, k - 1);
    reverse(nums, k, nums.length - 1);
}

// 旋转数组
private void reverse(int[] nums, int start, int end) {
    while (start < end) {
        // 记录开始位置的值
        int temp = nums[start];
        // 将结束位置的值赋给开始位置
        nums[start] = nums[end];
        // 将开始位置的值赋给结束位置
        nums[end] = temp;
        // 开始序列递增
        start++;
        // 结束序列递减
        end--;
    }
}

    示例：

    输入：l1 = [1,2,4], l2 = [1,3,4]
输出：[1,1,2,3,4,4]

输入：l1 = [], l2 = []
输出：[]

输入：l1 = [], l2 = [0]
输出：[0]

    提示：

    • 两个链表的节点数目范围是 [0, 50]
    • -100 <= Node.val <= 100
    • l1 和 l2 均按 非递减顺序 排列

    // 递归解法
public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
    // 如果链表1为null就返回链表2,如果链表2为null就返回链表1
    if (l1 == null || l2 == null) {
        return l2 == null ? l1 : l2;
    }
    // 如果链表1的值小于链表2的值
    if (l1.val < l2.val) {
        // 将链表1的指针指向链表2
        l1.next = mergeTwoLists(l1.next, l2);
        // 返回链表1
        return l1;
    } else {
        // 否则将链表2的指针指向链表1
        l2.next = mergeTwoLists(l1, l2.next);
        // 返回链表2
        return l2;
    }
}
    // 迭代解法
public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
    // 如果链表1为null就返回链表2,如果链表2为null就返回链表1
    if (l1 == null) {
        return l2;
    } else if (l2 == null) {
        return l1;
    }
    // 初始化开始节点
    ListNode dummy = new ListNode(0);
    // 将开始节点存入当前链表
    ListNode curr = dummy;
    // 如果两个链表都不为null，就对链表的值进行判断
    while (l1 != null && l2 != null) {
        // 如果链表1的值小于链表2的值
        if (l1.val <= l2.val) {
            // 就将当前链表的指针指向链表1
            curr.next = l1;
            // 将链表1的指针指向链表1
            l1 = l1.next;
        } else {
            // 否则将当前链表的指针指向链表2
            curr.next = l2;
            // 将链表2的指针指向链表2
            l2 = l2.next;
        }
        // 将当前链表的指针指向当前链表
        curr = curr.next;
    }
    // 如果当前链表的指针指向链表1且为null就指向链表2，否则指向链表1
    curr.next = l1 == null ? l2 : l1;
    // 返回开始元素指针指向的链表
    return dummy.next;
}

    题述：给你两个有序整数数组 nums1 和 nums2，请你将 nums2 合并到 nums1 中，使 nums1 成为一个有序数组。

    示例：

    输入：
nums1 = [1,2,3,0,0,0], m = 3
nums2 = [2,5,6],       n = 3

输出：[1,2,2,3,5,6]

输入：
nums1 = [1],             m = 1
nums2 = [],            n = 0
输出：[1]

    说明：

    初始化 nums1 和 nums2 的元素数量分别为 m 和 n 。
    你可以假设 nums1 有足够的空间（空间大小大于或等于 m + n）来保存 nums2 中的元素。

    提示：

    0 <= m, n <= 200
    1 <= m + n <= 200
    nums1.length == m + n
    nums2.length == n
    -109 <= nums1[i], nums2[i] <= 109

    // AC解法
public void merge(int[] nums1, int m, int[] nums2, int n) {
    // 记录数组1的末尾元素索引
    int endOne = m - 1;
    // 记录数组2的末尾元素索引
    int endTwo = n - 1;
    // 记录数组1和数组2衔接处的索引
    int midpoint = m + n - 1;
    // 如果两个数组的起始索引都大于零
    while (endOne >= 0 && endTwo >= 0) {
        // 如果数组1的末尾元素比数组2的大
        if (nums1[endOne] > nums2[endTwo]) {
            // 说明数组1的最大值是整个数组的最大值，所以将相邻之前的元素赋值给衔接处之前的位置
            nums1[midpoint--] = nums1[endOne--];
        } else {
            // 否则将数组2之前的元素赋给衔接处之前的位置
            nums1[midpoint--] = nums2[endTwo--];
        }
    }
    // 如果数组2的末尾索引一直大于0
    while (endTwo >= 0) {
        // 就依次将该所引之前的元素赋给数组1
        nums1[midpoint--] = nums2[endTwo--];
    }
}

}
    // 最佳解法
public void merge(int[] nums1, int m, int[] nums2, int n) {
    // 初始化三个节点的索引，数组1的末尾索引，数组1和数组2衔接处索引，数组2的末尾索引
    int tail1 = m - 1, tail2 = n - 1, finished = m + n - 1;
    // 如果数组1和数组2的末端索引值一直大于等于0
    while (tail1 >= 0 && tail2 >= 0) {

        nums1[finished--] = (nums1[tail1] > nums2[tail2]) ?
            nums1[tail1--] : nums2[tail2--];

    }
}

  5. 两数之和

    题述：给定一个整数数组 nums 和一个整数目标值 target，请你在该数组中找出 和为目标值 的那 两个 整数，并返回它们的数组下标。

    你可以假设每种输入只会对应一个答案。但是，数组中同一个元素不能使用两遍。

    你可以按任意顺序返回答案。

    示例：

    输入：nums = [2,7,11,15], target = 9
输出：[0,1]
解释：因为 nums[0] + nums[1] == 9 ，返回 [0, 1] 。

输入：nums = [3,2,4], target = 6
输出：[1,2]

输入：nums = [3,3], target = 6
输出：[0,1]

    提示：

    2 <= nums.length <= 103
    -109 <= nums[i] <= 109
    -109 <= target <= 109
    只会存在一个有效答案

    实现：

    // 一般方法
public int[] twoSum(int[] nums, int target) {
    // 初始化索引数组
    int[] result = new int[2];
    // 用于记录数组元素和其索引的映射关系
    Map<Integer, Integer> map = new HashMap<Integer, Integer>();
    // 遍历数组拿出每一个元素
    for (int i = 0; i < nums.length; i++) {
        // 如果集合的键是目标值减去当前元素
        if (map.containsKey(target - nums[i])) {
            // 将当前元素的索引赋给索引数组的第二个位置
            result[1] = i;
            // 根据该键获取集合元素的值赋给索引数组的第一个位置
            result[0] = map.get(target - nums[i]);

        map.put(nums[i], i);

    // 返回最新的索引数组
    return result;

    }

    return new int[]{};

        int p = i - 1;

public int trap(int[] height) {
    // 初始化数组长度
    int len = height.length;
    // 初始化左端索引
    int left = 0;
    // 初始化右端索引
    int right = len - 1;
    // 初始化结果值
    int res = 0;

    int maxleft = 0, maxright = 0;

            // 左端索引递增
            left++;

// 确定窗口的起点位置和终点位置；从起点位置一直找k个元素求出最大值输出(解不出来，超出时间限制)
    // 优化的暴力解法
public int[] maxSlidingWindow(int[] nums, int k) {
    final int[] max_left = new int[nums.length];
    final int[] max_right = new int[nums.length];

    max_left[0] = nums[0];
    max_right[nums.length - 1] = nums[nums.length - 1];

    for (int i = 1; i < nums.length; i++) {
        max_left[i] = (i % k == 0) ? nums[i] : Math.max(max_left[i - 1], nums[i]);

        final int j = nums.length - i - 1;
        max_right[j] = (j % k == 0) ? nums[j] : Math.max(max_right[j + 1], nums[j]);
    }

    final int[] sliding_max = new int[nums.length - k + 1];
    for (int i = 0, j = 0; i + k <= nums.length; i++) {
        sliding_max[j++] = Math.max(max_right[i], max_left[i + k - 1]);
    }

    return sliding_max;
}
    // 单调解法(队列)
public int[] maxSlidingWindow(int[] nums, int k) {
    if (nums == null || k <= 0) {
        return new int[0];
    }
    int n = nums.length;
    int[] r = new int[n - k + 1];
    int ri = 0;
    // store index
    Deque<Integer> q = new ArrayDeque<>();
    for (int i = 0; i < nums.length; i++) {
        // remove numbers out of range k
        while (!q.isEmpty() && q.peek() < i - k + 1) {
            q.poll();
        }
        // remove smaller numbers in k range as they are useless
        while (!q.isEmpty() && nums[q.peekLast()] < nums[i]) {
            q.pollLast();
        }
        // q contains index... r contains content
        q.offer(i);
        if (i >= k - 1) {
            r[ri++] = nums[q.peek()];
        }
    }
    return r;
}

    /** Initialize your data structure here. Set the size of the deque to be k. */
    public MyCircularDeque(int k) {
        this.q = new int[k + 1];
        this.head = this.count = 0;
        this.tail = 1;
        this.cap = k;
        this.size = k + 1;
    }

    /** Adds an item at the front of Deque. Return true if the operation is successful. */
    public boolean insertFront(int value) {
        if (isFull()) return false;
        this.q[this.head] = value;
        this.count++;
        this.head = (this.head - 1 + this.size) % this.size;
        return true;
    }

    /** Adds an item at the rear of Deque. Return true if the operation is successful. */
    public boolean insertLast(int value) {
        if (isFull()) return false;
        this.q[this.tail] = value;
        this.count++;
        this.tail = (this.tail + 1) % this.size;
        return true;
    }

    /** Deletes an item from the front of Deque. Return true if the operation is successful. */
    public boolean deleteFront() {
        if (isEmpty()) return false;
        this.head = (this.head + 1) % this.size;
        this.count--;
        return true;
    }

    /** Deletes an item from the rear of Deque. Return true if the operation is successful. */
    public boolean deleteLast() {
        if (isEmpty()) return false;
        this.tail = (this.tail - 1 + this.size) % this.size;
        this.count--;
        return true;
    }

    /** Get the front item from the deque. */
    public int getFront() {
        if (isEmpty()) return -1;
        int id = (this.head + 1) % this.size;
        return this.q[id];
    }

    /** Get the last item from the deque. */
    public int getRear() {
        if (isEmpty()) return -1;
        int id = (this.tail - 1 + this.size) % this.size;
        return this.q[id];
    }

    /** Checks whether the circular deque is empty or not. */
    public boolean isEmpty() {
        return this.count == 0;
    }

    /** Checks whether the circular deque is full or not. */
    public boolean isFull() {
        return this.count == this.cap;
    }
}

    // 集合解法
public int[] twoSum(int[] nums, int target) {
    // 创建两数集合
    HashMap<Integer, Integer> map = new HashMap<>();
    // 遍历两数集合
    for(int i = 0; i < nums.length; i++) {
        // 如果集合中包含目标值减去集合中任意元素的值
        if(map.containsKey(target - nums[i])) {
            // 则返回一个新数组，里面包含上面差值对应的索引，和当前元素的索引
            return new int[] {map.get(target - nums[i]), i};
        }
        else {
            // 否则向集合中添加当前元素
            map.put(nums[i], i);
        }
    }
    // 返回新的数组
    return new int[]{};
}

4.1 分治和回溯

  1. 本质

  4. 回溯

    采用试错的思想，尝试分步的去解决一个问题；在分步解决问题的过程中，当通过尝试发现现有的分步答案不能得到有效(正确)的解答的时候，它将取消上一步甚至是上几步的计算，再通过其他的可能的分步解答再次尝试寻找问题的答案。

    1. 通常使用最简单的递归方法来实现，在反复重复上述的步骤可能出现的两种情况

      1. 找了一个可能存在的正确答案
      2. 在尝试了所有可能的分步方法后宣告该问题没有答案

    2. 时间复杂度

      在最坏情况下，回溯算法会导致一次复杂度为指数的时间计算

    // 分治解法
public double myPow(double x, int n) {
    if(n == 0) {
        return 1;
    }

    if(n < 0) {
        return 1/x * myPow(1/x, -(n + 1));
    }
    return (n%2 == 0) ? myPow(x*x, n/2) : x*myPow(x*x, n/2);
}
    // 对数迭代法
public double myPow(double x, int n) {
    double ans = 1;
    long absN = Math.abs((long) n);
    while (absN > 0) {
        if ((absN & 1) == 1) {
            ans *= x;
        }
        absN >>= 1;
        x *= x;
    }
    return n < 0 ? 1 / ans : ans;
}

  3. 牛顿迭代法

    牛顿迭代法快速寻找平方根

    下面这种方法可以很有效地求出根号a的近似值：首先随便猜一个近似值x，然后不断令x等于x和a/x的平均数，迭代个六七次后x的值就已经相当精确了。

    例如，我想求根号2等于多少。假如我猜测的结果为4，虽然错的离谱，但你可以看到使用牛顿迭代法后这个值很快就趋近于根号2了：
    (       4  + 2/   4     ) / 2 = 2.25
(    2.25  + 2/   2.25  ) / 2 = 1.56944..
( 1.56944..+ 2/1.56944..) / 2 = 1.42189..
( 1.42189..+ 2/1.42189..) / 2 = 1.41423..
....

    这种算法的原理很简单，我们仅仅是不断用(x,f(x))的切线来逼近方程x^2-a=0的根。根号a实际上就是x^2-a=0的一个正实根，这个函数的导数是2x。也就是说，函数上任一点(x,f(x))处的切线斜率是2x。那么，x-f(x)/(2x)就是一个比x更接近的近似值。代入f(x)=x^2-a得到x-(x^2-a)/(2x)，也就是(x+a/x)/2。

    同样的方法可以用在其它的近似值计算中。[Quake III的源码](http://www.matrix67.com/blog/article.asp?id=411)中有一段非常牛B的开方取倒函数。
