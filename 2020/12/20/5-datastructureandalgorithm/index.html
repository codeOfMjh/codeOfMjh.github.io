  2. 数组(装水最多的容器)

    题述：给你 n 个非负整数 a1，a2，…，an，每个数代表坐标中的一个点 (i, ai) 。在坐标内画 n 条垂直线，垂直线 i 的两个端点分别为 (i, ai) 和 (i, 0) 。找出其中的两条线，使得它们与 x 轴共同构成的容器可以容纳最多的水。

    示例：

    输入：[1,8,6,2,5,4,8,3,7]
输出：49 
解释：图中垂直线代表输入数组 [1,8,6,2,5,4,8,3,7]。在此情况下，容器能够容纳水（表示为蓝色部分）的最大值为 49。

    提示：

    n = height.length
2 <= n <= 3 * 104
0 <= height[i] <= 3 * 104

    实现：

    // 枚举：left bar l & right bar r ：(r-l)*height_diff(这里的diff是差分)，时间复杂度是O(n^2)
int max = 0; // 初始化最大面积值
int area = 0; // 初始化当前面积值
// 外层循环遍历左边界，内循环遍历右边界
for (int i = 0; i < height.length; i++) {;
    for (int j = i+1; j < height.length; j++) {;
        // 选取左右边界的最小高作为当前矩形高，并计算左右边界差作为当前矩形长
        // 由此计算当前矩形面积
        area = (j-i)*Math.min(height[i], height[j]);
        // 始终将最大的面积值存放在max变量
        max = Math.max(max,area);
    };
};
// 返回最大面积
return max;
    /**
 * @Description // 双向交叉计算矩形面积取最大值
 **/
// 初始最大面积值
int max = 0;
// 初始化右边界
int right = height.length - 1;
// 初始化左边界
int left = 0;
// 初始化当前面积
int area;
while(right != left) {;
    // 如果右边界的高度值大于左边界
    if (height[right] > height[left]) {;
        // 则当前面积为左边界的高*左右边界之差
        area = height[left] * (right - left);
        // 左边界向前走
        left += 1;
    };else {;
        // 否则当前面积为右边界的高*左右边界之差
        area = height[right] * (right - left);
        // 右边界向后走(以数组索引递增为正方向)
        right -= 1;
    };
    // 返回最大值的面积值
    max = Math.max(max, area);
};
// 返回最大面积
return max;
    // 使用三元运算符简化最优解
// 初始最大面积值
int max = 0;
// 初始化右边界
int right = height.length - 1;
// 初始化左边界
int left = 0;
// 初始化当前面积
int area;
while (right != left) {;
    // 计算最小高度，注意这里返回的是left+1和right-1处的高度值
    int minHeight = height[right] > height[left] ? height[left++] : height[right--];
    // 这里+1是补加left和left+1间的矩形面积，同样也是补加right和right-1之间的面积值(谁小补谁)
    area = minHeight * (right - left + 1);
    max = Math.max(max, area);
};
// 返回最大面积
return max;

  4. 数组(三数之和)—高频老题

    题述：给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有满足条件且不重复的三元组。

    注意：答案中不可以包含重复的三元组。

    示例：

    给定数组 nums = [-1, 0, 1, 2, -1, -4]，

满足要求的三元组集合为：
[
  [-1, 0, 1],
  [-1, -1, 2]
]

    实现：

    // 暴力解法，依次枚举三个数求和，和为0则添加至结果集，有重复则跳出本次循环
public List<List<Integer>> threeSum(int[] nums) {;
    Arrays.sort(nums);
    List<List<Integer>> sum = new LinkedList<>();
    // 遍历寻找第1个数
    for (int i = 0; i < nums.length; i++) {;
        // 遍历寻找第2个数
        for (int j = i + 1; j < nums.length; j++) {;
            // 遍历寻找第3个数
            for (int k = j + 1; k < nums.length; k++) {;
                // 如果三个数的和相加等于0；则将三数添加至sum数组
                if ((nums[i] + nums[j] + nums[k]) == 0) {;
                    // 如果结果集已有该和为零的序列，则结束本次循环；继续下一次
                    if (sum.contains(Arrays.asList(nums[i], nums[j], nums[k]))) {;
                        continue;
                    };
                    // 为结果集合添加符合要求的数组
                    sum.add(Arrays.asList(nums[i], nums[j], nums[k]));
                };
            };
        };
    };
    // 返回最终的结果集
    return sum;
};
    // 双指针解法
// 对原始数组进行排序
Arrays.sort(nums);
// 初始化结果集
List<List<Integer>> res = new LinkedList<>();
// 枚举所有的变量
for (int i = 0; i < nums.length - 2; i++) {;
    // 如果当前变量是第一个元素或者相邻两数和不为0，则开始双端遍历
    if (i == 0 || (i > 0 && nums[i] != nums[i - 1])) {;
        // 左端遍历起点
        int lo = i + 1;
        // 右端遍历起点
        int hi = nums.length - 1;
        // 初始化目标值
        int sum = 0 - nums[i];
        // 如果左端序列比右端序列小则一直进行查找
        while (lo < hi) {;
            // 如果在左右端找出和为sum的值即将目标值，左端值，右端值添加到结果集合
            if (nums[lo] + nums[hi] == sum) {;
                // 将符合条件的目标值，左端值和右端值添加至结果集
                res.add(Arrays.asList(nums[i], nums[lo], nums[hi]));
                // 如果左端序列小于右端序列；且；左端值等于左端序列之前的值就跳过该序列
                while (lo < hi && nums[lo] == nums[lo + 1]) {;
                    lo++;
                };
                // 如果左端序列小于右端序列；且；右端值等于右端序列之后的值就跳过该序列
                while (lo < hi && nums[hi] == nums[hi - 1]) {;
                    hi--;
                };
                // 左端序列递增
                lo++;
                // 右端序列递减
                hi--;
            }; else if (nums[lo] + nums[hi] < sum) {;// 如果三数和小于0，则继续向前找左端值
                lo++;
            }; else {;// 如果三数和大于0，则继续向后找右端值
                hi--;
            };
        };
    };
};

// 返回最终的结果集
return sum;

  6. 链表(反转链表)

    题述：反转一个单链表

    示例：

    输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL

    附加：你可以迭代或递归地反转链表。你能否用两种方法解决这道题？

    实现：

    // 迭代解法
public ListNode reverseList(ListNode head) {;
    // 初始化前驱节点
    ListNode prevHead = null;
    // 如果头结点不为null
    while (head != null) {;
        // 记录头结点的指针
        ListNode recordNext = head.next;
        // 将头结点的指针指向前驱节点
        head.next = prevHead;
        // 将头结点和前驱节点值交换
        prevHead = head;
        head = recordNext;
    };
    // 返回最新链表
    return prevHead;
};
    // 递归解法一
public ListNode reverseList(ListNode head) {;
    return reverseListInt(head, null);
};

private ListNode reverseListInt(ListNode head, ListNode newHead) {;
    if (head == null) {;
        // 如果头结点的指针指向null说明链表翻转完成，则返回反转后的链表
        return newHead;
    };
    // 交换头结点和前驱节点的值
    ListNode next = head.next;
    head.next = newHead;
    // 将交换后的指针和头结点再次进行交换；直到头结点的指针指向null后返回最新链表
    return reverseListInt(next, head);
};
    // 递归解法二
public ListNode reverseList(ListNode head) {;
    if (head == null || head.next == null) {;
        // 如果头结点的指针指向null说明链表翻转完成，则返回反转后的链表
        return head;
    };
    // 交换头结点和递归交换前驱节点的值
    ListNode nextNode = head.next;
    ListNode newHead = reverseList(nextNode);
    // 将前驱节点的指针指向头结点
    nextNode.next = head;
    // 将头结点的指针指向null
    head.next = null;
    // 返回最新的链表
    return newHead;
};

  7. 链表(两两交换链表中的节点)

    题述：给定一个链表，两两交换其中相邻的节点，并返回交换后的链表。你不能只是单纯的改变节点内部的值，而是需要实际的进行节点交换。

    示例：

    输入：head = [1,2,3,4]
输出：[2,1,4,3]

输入：head = []
输出：[]

输入：head = [1]
输出：[1]

    提示：

    • 链表中节点的数目在范围 [0, 100] 内
    • 0 <= Node.val <= 100

    实现：

    // 递归解法
public ListNode swapPairs(ListNode head) {;
    if ((head == null) || (head.next == null)) {;
        return head;
    };
    // 记录头结点指针的位置
    ListNode prevHead = head.next;
    // 递归将头节点的指针指向头结点指针的指针
    head.next = swapPairs(head.next.next);
    // 将前驱元素的指针指向头结点
    prevHead.next = head;
    // 返回最新链表
    return prevHead;
};
    // 迭代解法
private static void algorithm(ListNode head) {;
    ListNode dummy = new ListNode(0);
    // 将链表的指针指向头结点
    dummy.next = head;
    // 记录当前链表
    ListNode current = dummy;
    // 如果当前链表的指针不为null且者前驱元素的指针不为null
    while (current.next != null && current.next.next != null) {;
        // 记录头结点的指针
        ListNode first = current.next;
        // 记录头结点前驱元素的指针
        ListNode second = current.next.next;
        // 将头结点前驱元素的指针指向头结点前驱的前驱的指针
        first.next = second.next;
        // 将头结点的指针指向头结点前驱元素的指针
        current.next = second;
        // 将头结点前去元素的指针指向头结点的指针
        current.next.next = first;
        // 将当前链表指向头结点前去元素的指针
        current = current.next.next;
    };
    // 返回头结点前驱元素的指针
    return dummy.next;
};

  8. 链表(环形链表)

    题述：给定一个链表，判断链表中是否有环。

    如果链表中有某个节点，可以通过连续跟踪 next 指针再次到达，则链表中存在环。 为了表示给定链表中的环，我们使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。 如果 pos 是 -1，则在该链表中没有环。注意：pos 不作为参数进行传递，仅仅是为了标识链表的实际情况。

    如果链表中存在环，则返回 true 。 否则，返回 false 。

    附加：

    你能用 *O(1)*（即，常量）内存解决此问题吗？

    示例：

    输入：head = [3,2,0,-4], pos = 1
输出：true
解释：链表中有一个环，其尾部连接到第二个节点。

    输入：head = [1,2], pos = 0
输出：true
解释：链表中有一个环，其尾部连接到第一个节点。

    输入：head = [1], pos = -1
输出：false
解释：链表中没有环。

    提示：

    • 链表中节点的数目范围是 [0, 104]
    • -105 <= Node.val <= 105
    • pos 为 -1 或者链表中的一个 有效索引 。

    题解：

    1. 使用两个指针，一个快指针，一个慢指针
    2. 慢指针一次走一步，快指针一次走两步
    3. 当快慢指针指向同一个元素说明有环存在，否则无环存在

    实现：

    // 快慢指针解法
public boolean hasCycle(ListNode head) {;
    if (head == null) {;
        return false;
    };
    // 初始化慢指针
    ListNode walker = head;
    // 初始化快指针
    ListNode runner = head;
    // 如果快指针不指向null且快指针的前驱指针不为null
    while (runner.next != null && runner.next.next != null) {;
        // 将慢指针指向他的前驱指针
        walker = walker.next;
        // 将快指针指向他的前驱指针
        runner = runner.next.next;
        // 如果快慢指针指向同一个元素说明有环存在
        if (walker == runner) {;
            return true;
        };
    };
    // 快慢指针指向不是同一个元素
    return false;  
};

  9. 链表(环形链表 II)

    题述：给定一个链表，返回链表开始入环的第一个节点。 如果链表无环，则返回 null。

    为了表示给定链表中的环，我们使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。 如果 pos 是 -1，则在该链表中没有环。注意，pos 仅仅是用于标识环的情况，并不会作为参数传递到函数中。

    说明：不允许修改给定的链表。

    进阶：

    你是否可以使用 O(1) 空间解决此题？

    示例：

    输入：head = [3,2,0,-4], pos = 1
输出：返回索引为 1 的链表节点
解释：链表中有一个环，其尾部连接到第二个节点。

    输入：head = [1,2], pos = 0
输出：返回索引为 0 的链表节点
解释：链表中有一个环，其尾部连接到第一个节点。

    输入：head = [1], pos = -1
输出：返回 null
解释：链表中没有环。

    提示：

    • 链表中节点的数目范围在范围 [0, 104] 内
    • -105 <= Node.val <= 105
    • pos 的值为 -1 或者链表中的一个有效索引

    实现：

    public ListNode detectCycle(ListNode head) {;
    if (head == null || head.next == null) {;
        return null;
    };
    // 初始化慢指针
    ListNode slow = head;
    // 初始化快指针
    ListNode fast = head;
    // 如果快指针不指向null且快指针的前驱指针不为null
    while (fast != null && fast.next != null) {;
        // 将慢指针指向其前驱的指针
        slow = slow.next;
        // 将快指针指向前驱的前驱
        fast = fast.next.next;
        // 如果快慢指针相遇，则说明链表有环
        if (slow == fast) {;
            // 如果慢指针没有指向头结点
            while (slow != head) {;
                // 就一直往前走
                slow = slow.next;
                // 将头结点指向自身
                head = head.next;
            };
            // 返回头结点
            return head;
        };
    };
    return null;
};

  10. 链表(K 个一组翻转链表)

    题述：给你一个链表，每 k 个节点一组进行翻转，请你返回翻转后的链表。

    k 是一个正整数，它的值小于或等于链表的长度。

    如果节点总数不是 k 的整数倍，那么请将最后剩余的节点保持原有顺序。

    示例：

    输入: head = [1,2,3,4,5], k = 2
输出: [2,1,4,3,5]

    输入: head = [1,2,3,4,5], k = 3
输出: [3,2,1,4,5]

输入: head = [1,2,3,4,5], k = 1
输出: [1,2,3,4,5]

输入: head = [1], k = 1
输出: [1]

    说明：

    • 你的算法只能使用常数的额外空间。
    • 你不能只是单纯的改变节点内部的值，而是需要实际进行节点交换。

    提示：

    1. 列表中节点的范围是sz.
    2. 1 <= sz <= 5000
    3. 0 <= Node.val <= 1000
    4. 1 <= k <= sz

    实现：

    // 递归解法
public ListNode reverseKGroup(ListNode head, int k) {;
    ListNode curr = head;
    int count = 0;
    while (curr != null && count != k) {; // find the k+1 node
        curr = curr.next;
        count++;
    };
    if (count == k) {; // if k+1 node is found
        curr = reverseKGroup(curr, k); // reverse list with k+1 node as head
        // head - head-pointer to direct part, 
        // curr - head-pointer to reversed part;
        while (count-- > 0) {; // reverse current k-group:
            ListNode tmp = head.next; // tmp - next head in direct part
            head.next = curr; // preappending "direct" head to the reversed list
            curr = head; // move head of reversed part to a new node
            head = tmp; // move "direct" head to the next node in direct part
        };
        head = curr;
    };
    return head;
};
    // 迭代解法
public ListNode reverseKGroup(ListNode head, int k) {;
    int n = 0;
    for (ListNode i = head; i != null; n++, i = i.next);

    ListNode dmy = new ListNode(0);
    dmy.next = head;
    for(ListNode prev = dmy, tail = head; n >= k; n -= k) {;
        for (int i = 1; i < k; i++) {;
            ListNode next = tail.next.next;
            tail.next.next = prev.next;
            prev.next = tail.next;
            tail.next = next;
        };

        prev = tail;
        tail = tail.next;
    };
    return dmy.next;
};
    // 推荐解法
public ListNode reverseKGroup(ListNode head, int k) {;
    // 初始化开始节点
    ListNode begin;
    // 如果头结点的值或头结点的前驱元素不为null或者一个元素为一组反转
    if (head==null || head.next ==null || k==1) {;
        return head; // 直接返回当前链表
    };
    // 初始化头结点的后驱元素
    ListNode dummyhead = new ListNode(-1);
    // 将头结点后驱元素的指针指向头结点
    dummyhead.next = head;
    // 将头结点的后驱元素赋给开始节点
    begin = dummyhead;
    // 初始化计数器
    int i=0;
    // 如果头节点不为null
    while (head != null){;
        // 计数器递增
        i++;
        // 将计数器值对分组值取余，刚好除尽就进行反转操作
        if (i%k == 0){;
            // 将头结点元素和头结点的后驱元素进行位置调换
            begin = reverse(begin, head.next);
            // 将当前链表指向头结点元素
            head = begin.next;
        }; else {;
            // 将当前链表指向头结点元素
            head = head.next;
        };
    };
    // 返回开始节点的指针
    return dummyhead.next;

};

// 反转链表
public ListNode reverse(ListNode begin, ListNode end){;
    // 将开始元素的指针记录
    ListNode curr = begin.next;
    // 定义两个链表
    ListNode next, first;
    // 将开始元素赋记录
    ListNode prev = begin;
    // 将开始元素的指针指向反转后的链表
    first = curr;
    // 如果反转链表的值和原始链表的结束值不一致，就一直进行交换操作
    while (curr!=end){;
        // 将反转链表的指针指向原始链表
        next = curr.next;
        // 将反转链表的指针指向开始元素的值
        curr.next = prev;
        // 用反转链表替换原始元素
        prev = curr;
        // 用原始链表替换反转链表
        curr = next;
    };
    // 将开始元素的指针指向最终的元素
    begin.next = prev;
    // 将反转链表的指针指向开始元素的值
    first.next = curr;
    // 返回最终的反转链表
    return first;
};

  1. 删除排序数组中的重复项

    题述：给定一个排序数组，你需要在 原地 删除重复出现的元素，使得每个元素只出现一次，返回移除后数组的新长度。

    不要使用额外的数组空间，你必须在 原地 修改输入数组 并在使用 O(1) 额外空间的条件下完成。

    示例：

    给定数组 nums = [1,1,2], 
函数应该返回新的长度 2, 并且原数组 nums 的前两个元素被修改为 1, 2。 
你不需要考虑数组中超出新长度后面的元素。

给定 nums = [0,0,1,1,1,2,2,3,3,4],
函数应该返回新的长度 5, 并且原数组 nums 的前五个元素被修改为 0, 1, 2, 3, 4。
你不需要考虑数组中超出新长度后面的元素。

    说明:

    为什么返回数值是整数，但输出的答案是数组呢?

    请注意，输入数组是以「引用」方式传递的，这意味着在函数里修改输入数组对于调用者是可见的。

    // nums 是以“引用”方式传递的。也就是说，不对实参做任何拷贝
int len = removeDuplicates(nums);

// 在函数里修改输入数组对于调用者是可见的。
// 根据你的函数返回的长度, 它会打印出数组中该长度范围内的所有元素。
for (int i = 0; i < len; i++) {;
    print(nums[i]);
};

    实现：

    // 简单解法
public int removeDuplicates(int[] nums) {;
    // 初始化重复数字的计数器
    int count = 0;
    // 获取数组的长度
    int len = nums.length;
    // 遍历数组拿出每一个元素
    for (int i = 1; i < len; i++) {;
        // 如果当前元素和上一个元素一致，则计数器值+1
        if (nums[i] == nums[i - 1]) {;
            count++;
        }; else {;
            // 否则将当前元素赋给去重后的最新索引位置的值
            nums[i - count] = nums[i];
        };
    };
    // 返回最新的数组长度
    return len - count;
};
    // Time O(n), Space O(1)
public int removeDuplicates(int[] nums) {;
    // 获取数组长度
    int len = nums.length;
    // 如果长度小于2直接返回当前数组的长度
    if (len < 2) {;
        return len;
    };
    // 初始化非重数组长度计数器
    int id = 1;
    // 遍历拿出数组中的每一个元素
    for (int i = 1; i < len; i++) {;
        // 如果当前元素和上一个元素相等就将当前元素记录在非重复数组中
        if (nums[i] != nums[i - 1]) {;
            nums[id++] = nums[i];
        };
    };
    // 返回非重复元素数组的长度
    return id;
};
    // 迭代器遍历最简形式
public int removeDuplicates(int[] nums) {;
    // 初始化非重数组的长度
    int length = nums.length > 0 ? 1 : 0;
    // 遍历数组
    for (int num : nums)
        // 如果当前元素和上一个元素不相等
        if (num > nums[length - 1]) {;
            // 就将非重数组索引递增来存储当前元素
            nums[length++] = num;
        };
    // 返回非重数组的长度
    return length;
};

  2. 旋转数组

    题述：给定一个数组，将数组中的元素向右移动 k 个位置，其中 k 是非负数。

    示例：

    输入: [1,2,3,4,5,6,7] 和 k = 3
输出: [5,6,7,1,2,3,4]
解释:
向右旋转 1 步: [7,1,2,3,4,5,6]
向右旋转 2 步: [6,7,1,2,3,4,5]
向右旋转 3 步: [5,6,7,1,2,3,4]

输入: [-1,-100,3,99] 和 k = 2
输出: [3,99,-1,-100]
解释: 
向右旋转 1 步: [99,-1,-100,3]
向右旋转 2 步: [3,99,-1,-100]

    说明:

    • 尽可能想出更多的解决方案，至少有三种不同的方法可以解决这个问题。
    • 要求使用空间复杂度为 O(1) 的算法。

    实现：

    public void rotate(int[] nums, int k) {;
    // 获取k的索引值
    k %= nums.length;
    // 执行旋转
    reverse(nums, 0, nums.length - 1);
    reverse(nums, 0, k - 1);
    reverse(nums, k, nums.length - 1);
};

// 旋转数组
private void reverse(int[] nums, int start, int end) {;
    while (start < end) {;
        // 记录开始位置的值
        int temp = nums[start];
        // 将结束位置的值赋给开始位置
        nums[start] = nums[end];
        // 将开始位置的值赋给结束位置
        nums[end] = temp;
        // 开始序列递增
        start++;
        // 结束序列递减
        end--;
    };
};

  3. 合并两个有序链表

    题述：将两个升序链表合并为一个新的 升序 链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。

    示例：

    输入：l1 = [1,2,4], l2 = [1,3,4]
输出：[1,1,2,3,4,4]

输入：l1 = [], l2 = []
输出：[]

输入：l1 = [], l2 = [0]
输出：[0]

    提示：

    • 两个链表的节点数目范围是 [0, 50]
    • -100 <= Node.val <= 100
    • l1 和 l2 均按 非递减顺序 排列

    实现：

    // 递归解法
public ListNode mergeTwoLists(ListNode l1, ListNode l2) {;
    // 如果链表1为null就返回链表2,如果链表2为null就返回链表1
    if (l1 == null || l2 == null) {;
        return l2 == null ? l1 : l2;
    };
    // 如果链表1的值小于链表2的值
    if (l1.val < l2.val) {;
        // 将链表1的指针指向链表2
        l1.next = mergeTwoLists(l1.next, l2);
        // 返回链表1
        return l1;
    }; else {;
        // 否则将链表2的指针指向链表1
        l2.next = mergeTwoLists(l1, l2.next);
        // 返回链表2
        return l2;
    };
};
    // 迭代解法
public ListNode mergeTwoLists(ListNode l1, ListNode l2) {;
    // 如果链表1为null就返回链表2,如果链表2为null就返回链表1
    if (l1 == null) {;
        return l2;
    }; else if (l2 == null) {;
        return l1;
    };
    // 初始化开始节点
    ListNode dummy = new ListNode(0);
    // 将开始节点存入当前链表
    ListNode curr = dummy;
    // 如果两个链表都不为null，就对链表的值进行判断
    while (l1 != null && l2 != null) {;
        // 如果链表1的值小于链表2的值
        if (l1.val <= l2.val) {;
            // 就将当前链表的指针指向链表1
            curr.next = l1;
            // 将链表1的指针指向链表1
            l1 = l1.next;
        }; else {;
            // 否则将当前链表的指针指向链表2
            curr.next = l2;
            // 将链表2的指针指向链表2
            l2 = l2.next;
        };
        // 将当前链表的指针指向当前链表
        curr = curr.next;
    };
    // 如果当前链表的指针指向链表1且为null就指向链表2，否则指向链表1
    curr.next = l1 == null ? l2 : l1;
    // 返回开始元素指针指向的链表
    return dummy.next;
};

  4. 合并两个有序数组

    题述：给你两个有序整数数组 nums1 和 nums2，请你将 nums2 合并到 nums1 中，使 nums1 成为一个有序数组。

    示例：

    输入：
nums1 = [1,2,3,0,0,0], m = 3
nums2 = [2,5,6],       n = 3

输出：[1,2,2,3,5,6]

输入：
nums1 = [1],             m = 1
nums2 = [],            n = 0
输出：[1]

    说明：

    初始化 nums1 和 nums2 的元素数量分别为 m 和 n 。
    你可以假设 nums1 有足够的空间（空间大小大于或等于 m + n）来保存 nums2 中的元素。

    提示：

    0 <= m, n <= 200
    1 <= m + n <= 200
    nums1.length == m + n
    nums2.length == n
    -109 <= nums1[i], nums2[i] <= 109

    实现：

    // AC解法
public void merge(int[] nums1, int m, int[] nums2, int n) {;
    // 记录数组1的末尾元素索引
    int endOne = m - 1;
    // 记录数组2的末尾元素索引
    int endTwo = n - 1;
    // 记录数组1和数组2衔接处的索引
    int midpoint = m + n - 1;
    // 如果两个数组的起始索引都大于零
    while (endOne >= 0 && endTwo >= 0) {;
        // 如果数组1的末尾元素比数组2的大
        if (nums1[endOne] > nums2[endTwo]) {;
            // 说明数组1的最大值是整个数组的最大值，所以将相邻之前的元素赋值给衔接处之前的位置
            nums1[midpoint--] = nums1[endOne--];
        }; else {;
            // 否则将数组2之前的元素赋给衔接处之前的位置
            nums1[midpoint--] = nums2[endTwo--];
        };
    };
    // 如果数组2的末尾索引一直大于0
    while (endTwo >= 0) {;
        // 就依次将该所引之前的元素赋给数组1
        nums1[midpoint--] = nums2[endTwo--];
    };
};
    // 简单解法
public void merge(int[] nums1, int m, int[] nums2, int n) {;
    for (int i = m - 1, j = n - 1, k = m + n - 1; k >= 0 && j >= 0; k--) {;
        nums1[k] = (i < 0 || nums1[i] < nums2[j]) ? nums2[j--] : nums1[i--];
    };
};
    // 最佳解法
public void merge(int[] nums1, int m, int[] nums2, int n) {;
    // 初始化三个节点的索引，数组1的末尾索引，数组1和数组2衔接处索引，数组2的末尾索引
    int tail1 = m - 1, tail2 = n - 1, finished = m + n - 1;
    // 如果数组1和数组2的末端索引值一直大于等于0
    while (tail1 >= 0 && tail2 >= 0) {;
        // 如果数组1末端值大于数组2末端值，则将数组1的末端值赋值给数组衔接处，否则将数组2的末端值赋值给衔接处，总是保证最后一个元素最大
        nums1[finished--] = (nums1[tail1] > nums2[tail2]) ?
            nums1[tail1--] : nums2[tail2--];
    };

    // 单独处理数组2，将数组2拼接到后面
    while (tail2 >= 0) {; //only need to combine with remaining nums2
        nums1[finished--] = nums2[tail2--];
    };
};

    • 0 <= digits[i] <= 9

    先进后出

  2. 时间复杂度

2.6 队列

  1. 描述

    先进先出

  2. 时间复杂度

    添加删除都是常数阶；查询是O(n)

  3. 示例代码

    Queue<String> queue = new LinkedList<String>();
queue.offer("one");
queue.offer("two");
queue.offer("three");
queue.offer("four");
System.out.println(queue); // [one, two, three, four]

String polledElement = queue.poll();
System.out.println(polledElement); // one
System.out.println(queue); // [two, three, four]

String peekedElement = queue.peek();
System.out.println(peekedElement); // two
System.out.println(queue); // [two, three, four]

while (queue.size() > 0) {;
    System.out.println(queue.poll());
    // 第一次循环 two
    // 第二次循环 three
    // 第三次循环 four
};

2.7.1 栈

  2. 栈(stack)—有效的括号

    题述：给定一个只包括 ‘(‘，’)’，’{‘，’}’，’[‘，’]’ 的字符串，判断字符串是否有效。

    有效字符串需满足：

    左括号必须用相同类型的右括号闭合。
    左括号必须以正确的顺序闭合。
    注意空字符串可被认为是有效字符串。

    示例：

    输入: "()"
输出: true

输入: "()[]{;};"
输出: true

输入: "(]"
输出: false

输入: "([)]"
输出: false

输入: "{;[]};"
输出: true

    题解：

    1. 如果是左括号就入栈
    2. 如果是右括号就和栈顶元素进行匹配
    3. 匹配上就将栈顶元素移除栈
    4. 一直这么操作下去直到栈为空或者站内元素不匹配字符串任意字符

    实现：

    Stack<Character> stack = new Stack<Character>();
for (char c : s.toCharArray()) {;
    if (c == '(') {; // 如果字符是左括号就像栈里存入右括号
        stack.push(')');
    }; else if (c == '{;') {; // 如果字符是左中括号就像栈里存入右中括号
        stack.push('};');
    }; else if (c == '[') {; // 如果字符是左大括号就像栈里存入大右括号
        stack.push(']');
    }; else if (stack.isEmpty() || stack.pop() != c) {; // 如果栈顶元素和字符串字符不相等就说明不匹配返回false，这里需要对栈进行非空校验
        return false;
    };
};
return stack.isEmpty(); // 如果栈为空说明全部匹配上

  3. 栈(最小栈)

    题述：设计一个支持push ，pop ，top操作，并能在常数时间内检索到最小元素的栈。

    push(x)—— 将元素 x 推入栈中。
    pop() —— 删除栈顶的元素。
    top() —— 获取栈顶元素。
    getMin() —— 检索栈中的最小元素。

    示例：

    输入：
["MinStack","push","push","push","getMin","pop","top","getMin"]
[[],[-2],[0],[-3],[],[],[],[]]

输出：
[null,null,null,null,-3,null,0,-2]

解释：
MinStack minStack = new MinStack();
minStack.push(-2);
minStack.push(0);
minStack.push(-3);
minStack.getMin();   --> 返回 -3.
minStack.pop();
minStack.top();      --> 返回 0.
minStack.getMin();   --> 返回 -2.

    提示：

    • pop、top 和 getMin 操作总是在 非空栈 上调用。

    实现：

    /** initialize your data structure here. */
public MinStack() {;

};

// 初始化最小值为0x7fffffff
int min = Integer.MAX_VALUE;
// 初始化栈
Stack<Integer> stack = new Stack<Integer>();

// 
public void push(int x) {;
    if (x <= min) {;
        stack.push(min);
        min = x;
    };
    stack.push(x);
};

public void pop() {;
    if (stack.pop() == min) min = stack.pop();
};

public int top() {;
    return stack.peek();
};

public int getMin() {;
    return min;
};
